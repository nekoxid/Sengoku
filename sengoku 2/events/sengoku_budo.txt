# Tournament is over
province_event = {
	id = BUDO.252
	title = "BUDOTITLE"
	desc = "EVTDESC_BUDO_252"
	picture = "GFX_evt_joust"

	trigger = {
		had_province_flag = {
			flag = tournament_60_days_budo
			days = 60
		}
		owner = {
			has_character_flag = winner_budo
			has_character_flag = second_place_budo
			has_character_flag = third_place_budo
		}
	}

	mean_time_to_happen = {
		days = 1
	}

	option = {
		name = "EVTOPTA88252"
		clr_province_flag = tournament_60_days_budo
		owner = {
			clr_character_flag = tournament_begins_budo
			clr_character_flag = winner_budo
			clr_character_flag = second_place_budo
			clr_character_flag = third_place_budo
			clr_character_flag = attending_tournament_budo
			clr_character_flag = attending_tournament_event_budo
			clr_character_flag = tournament_third_place_budo
			clr_character_flag = tournament_second_place_budo
			clr_character_flag = tournament_winner_budo
			set_character_flag = tournament_has_ended_budo
			any_realm_character = {
				limit = {
					has_character_flag = attending_tournament_budo
					OR = {
						has_character_flag = tournament_third_place_budo
						has_character_flag = tournament_second_place_budo
						has_character_flag = tournament_winner_budo
					}
				}
				set_character_flag = tournament_has_ended_budo
				clr_character_flag = tournament_third_place_budo
				clr_character_flag = tournament_second_place_budo
				clr_character_flag = tournament_winner_budo
				opinion = {
					modifier = opinion_tournament_participant
					who = PREV
					years = 2
				}
				clr_character_flag = attending_tournament_budo
				clr_character_flag = attending_tournament_event_budo
			}
			any_realm_character = {
				limit = {
					has_character_flag = attending_tournament_budo
					NOT = {
						character = PREV
					}
					NOT = {
						has_character_flag = tournament_third_place_budo
					}
					NOT = {
						has_character_flag = tournament_second_place_budo
					}
					NOT = {
						has_character_flag = tournament_winner_budo
					}
				}
				hidden_tooltip = {
					random_list = {
						10 = {
							character_event = { id = BUDO.253 }
						}
						10 = {
							character_event = { id = BUDO.254 }
						}
						10 = {
							character_event = { id = BUDO.255 }
						}
						10 = {
							character_event = { id = BUDO.256 }
						}
						10 = {
							character_event = { id = BUDO.257 }
						}
						10 = {
							character_event = { id = BUDO.258 }
						}
					}
				}
				set_character_flag = tournament_has_ended_budo
				opinion = {
					modifier = opinion_tournament_participant
					who = PREV
					years = 2
				}
				clr_character_flag = attending_tournament_budo
				clr_character_flag = attending_tournament_event_budo
			}
		}
	}
}

	only_men = yes

	prisoner = no

	trigger = {

		}
		top_liege = {
			has_character_flag = tournament_begins_budo
			NOT = {
				has_character_flag = tournament_has_ended_budo
			}
		}
	}

	mean_time_to_happen = {
		months = 1000
		modifier = {
			factor = 1.25
			martial = 8
		}
		modifier = {
			factor = 1.25
			martial = 10
		}
		modifier = {
			factor = 1.25
			martial = 12
		}
		modifier = {
			factor = 1.25
			martial = 14
		}
		modifier = {
			factor = 0.75
			NOT = { martial = 7 }
		}
		modifier = {
			factor = 0.75
			NOT = { martial = 5 }
		}
		modifier = {
			factor = 0.75
			NOT = { martial = 3 }
		}
		modifier = {
			factor = 0.75
			NOT = { martial = 1 }
		}
		modifier = {
			factor = 2
			trait = craven
		}
		modifier = {
			factor = 0.5
			trait = brave
		}
		modifier = {
			factor = 0.5
			trait = maimed
		}
	}

	immediate = {
		clr_character_flag = attending_tournament_budo
	}

	option = {
		name = "EVTOPTA88259"
		death = {
			death_reason = death_battle
		}
		hidden_tooltip = {
			if = {
				limit = {
					top_liege = {
						NOT = {
							character = ROOT
						}
					}
				}
				top_liege = { character_event = { id = BUDO.288  } }
			}
		}
	}
}
