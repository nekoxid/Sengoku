# AI Challenges
character_event = {
	id = SENGOKU_LIFE.900
	hide_window = yes
	
	ai = yes
	min_age = 16
	max_age = 60
	capable_only = yes
	prisoner = no
	
	trigger = {
		war = no
		primary_title = { higher_tier_than = COUNT }
		OR = {
			liege = {
				NOT = { has_character_flag = no_more_go }
				ai = no
				OR = {
					culture_group = asian
					trait = go_player
				}
				is_female = no
				prisoner = no
				age = 16
				NOT = { age = 65 }
				NOT = { trait = infirm }
				NOT = { trait = incapable }
			}
			liege = {
				any_realm_lord = {
					ai = no
					NOT = { has_character_flag = no_more_go }
					same_liege = ROOT
					OR = {
						culture_group = asian
						trait = go_player
					}
					is_female = no
					prisoner = no
					age = 16
					NOT = { age = 65 }
					NOT = { trait = infirm }
					NOT = { trait = incapable }
				}
			}
		}
		OR = {
			culture_group = asian
			trait = go_player
		}
		is_female = no
	}
	
	mean_time_to_happen = {
		months = 1400
	}
	
	immediate = {
		if = {
			limit = {
				liege = {
					OR = {
						culture_group = asian
						trait = go_player
					}
					ai = no
				}
			}
			liege = {
				character_event = { id = SENGOKU_LIFE.901 }
			}
		}
		if = {
			limit = {
				liege = {
					OR = {
						culture_group = asian
						trait = go_player
					}
					any_realm_lord = { ai = no }
				}
			}
			liege = {
				random_realm_lord = {
					limit = { ai = no }
					character_event = { id = SENGOKU_LIFE.901 }
				}
			}
		}		
	}
	
	option = {
		name = OK
	}
}

					50 = {
						character_event = { id = SENGOKU_LIFE.902 }

					}
				}
			}
			if = {
				limit = { trait = wroth }

			}
		}
	}

		name = EVTOPTBSENGOKU_LIFE.901

}
